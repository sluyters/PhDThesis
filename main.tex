\newcommand*\emptycirc[1][1ex]{
  \begin{tikzpicture}[baseline=(base)]
    \draw[circlecolor2, thick] (0,0) circle (#1);
    \node (base) at (0,-.7ex) {};
  \end{tikzpicture}
} 

\newcommand*\halfcirc[1][1ex]{%
  \begin{tikzpicture}[baseline=(base)]
    \draw[circlecolor1,fill=circlecolor1] (0,0)-- (90:#1) arc (90:270:#1) -- cycle ;
    \draw[circlecolor1,thick] (0,0) circle (#1);
    \node (base) at (0,-.7ex) {};
  \end{tikzpicture}
}

    \draw[circlecolor1,fill=circlecolor1,thick] (0,0) circle (#1);

  \end{tikzpicture}
}
